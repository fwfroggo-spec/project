// Function to pick a random agent
function pickRandomAgent() {
    const randomIndex = Math.floor(Math.random() * agents.length);  // Get a random index
    const agent = agents[randomIndex];  // Get the agent based on the index

    // Update the HTML with the chosen agent's name and image
    document.getElementById("agentName").textContent = agent.name;
    document.getElementById("agentImage").src = agent.imgSrc;
    document.getElementById("agentImage").alt = agent.name;
}

// Add event listener to the button
document.getElementById('randomizeBtn').addEventListener('click', pickRandomAgent);
